    \column{0.5}
    \block{1. Project Overview}{
        The \textbf{Makeitalive} project explores the transformation of a static landscape photograph into a realistic animation. 
        We preserve the original texture while injecting coherent movement.
        
        \vspace{1em}
        \textbf{Challenges met:}
        \begin{itemize}
            \item Automatic animation.
            \item HD temporal coherence.
            \item Realism of natural textures.
        \end{itemize}
    }
    
    \block{2. The Motion Flow Concept}{
        The approach is based on estimating an \textit{Optical Flow} field:
        \begin{itemize}
            \item Prediction of $(dx, dy)$ vectors per pixel.
            \item \textbf{Backward Mapping} for rendering:
            \[ I_{anim}(x, y, t) = I_{orig}(x - \alpha_t dx, y - \alpha_t dy) \]
        \end{itemize}
    }

    \block{3. Motion Flow Architecture (Custom U-Net)}{
        The U-Net architecture is optimized for flow prediction:
        
        \begin{center}
        \begin{tikzpicture}[node distance=1.5cm, transform shape, scale=0.8]
            \tikzstyle{layer} = [rectangle, draw, fill=blue!20, text width=2.5cm, text centered, minimum height=0.8cm]
            \tikzstyle{flow} = [->, >=stealth, thick]

            \node (in) [layer, fill=green!20] {Input Image};
            \node (enc1) [layer, below of=in] {Enc 32};
            \node (enc2) [layer, below of=enc1] {Enc 64};
            \node (enc3) [layer, below of=enc2] {Enc 128};
            \node (btl) [layer, below of=enc3, fill=orange!20] {Bottleneck 256};
            
            \node (dec3) [layer, right=3cm of enc3] {Dec 128};
            \node (dec2) [layer, right=3cm of enc2] {Dec 64};
            \node (dec1) [layer, right=3cm of enc1] {Dec 32};
            \node (out) [layer, right=3cm of in, fill=red!20] {Flow Output};

            \draw [flow] (in) -- (enc1); \draw [flow] (enc1) -- (enc2);
            \draw [flow] (enc2) -- (enc3); \draw [flow] (enc3) -- (btl);
            \draw [flow] (btl) -| (dec3); \draw [flow] (dec3) -- (dec2);
            \draw [flow] (dec2) -- (dec1); \draw [flow] (dec1) -- (out);

            \draw [dashed, flow, gray] (enc1) -- (dec1);
            \draw [dashed, flow, gray] (enc2) -- (dec2);
            \draw [dashed, flow, gray] (enc3) -- (dec3);
        \end{tikzpicture}
        \end{center}
        
        The model is extremely lightweight ($<100$MB) and allows for real-time inference on GPU.
    }

    \column{0.5}
    \block{4. Stable Video Diffusion (SVD) Approach}{
        \textbf{Stable Video Diffusion (SVD)}:
        \begin{itemize}
            \item 3D Latent Diffusion Model.
            \item \textbf{LoRA}: Lightweight fine-tuning for landscape textures (water, wind in trees).
            \item Superior visual quality for complex scenes thanks to iterative denoising.
        \end{itemize}
    }

    \block{5. Comparison of Techniques}{
        \begin{center}
        \begin{tabular}{|l|c|c|}
            \hline
            \textbf{Criterion} & \textbf{Motion Flow} & \textbf{SVD + LoRA} \\
            \hline
            Fidelity & Maximum & High \\
            Speed & Instant & Several seconds \\
            Weight & Light & Heavy \\
            Resources & CPU/GPU & High-End GPU \\
            \hline
        \end{tabular}
        \end{center}
    }

    \block{6. Result Evaluation}{
        Tests show that \textbf{Motion Flow} is ideal for subtle and efficient animations, while \textbf{SVD} provides unmatched depth of motion for longer videos.
    }

    \block{7. Conclusion}{
        The hybridization of the two delivers the best of both worlds: structural speed and generative textual richness.
    }
